#!/bin/bash

set -e

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
ROOT_DIR="$(cd "$SCRIPT_DIR/.." && pwd)"
CERTS_DIR="${ROOT_DIR}/.certs"
TEMPLATE="${SCRIPT_DIR}/manifests/webhook/template.yaml"
OUTPUT="${SCRIPT_DIR}/manifests/cluster-config/mutating.config.yaml"

rm -rf "${CERTS_DIR}"
mkdir -p "${CERTS_DIR}"

WEBHOOK_NAME="${WEBHOOK_NAME:-gitlab-workspaces-kubernetes-webhook}"
WEBHOOK_NAMESPACE="${WEBHOOK_NAMESPACE:-default}"

# Generate CA and server certs
openssl genrsa -out "${CERTS_DIR}/ca.key" 2048

openssl req -new -x509 -days 365 -key "${CERTS_DIR}/ca.key" \
  -subj "/C=AU/CN=${WEBHOOK_NAME}" \
  -out "${CERTS_DIR}/ca.crt"

openssl req -newkey rsa:2048 -nodes -keyout "${CERTS_DIR}/server.key" \
  -subj "/C=AU/CN=${WEBHOOK_NAME}" \
  -out "${CERTS_DIR}/server.csr"

openssl x509 -req \
  -extfile <(printf "subjectAltName=DNS:${WEBHOOK_NAME}.${WEBHOOK_NAMESPACE}.svc") \
  -days 365 \
  -in "${CERTS_DIR}/server.csr" \
  -CA "${CERTS_DIR}/ca.crt" -CAkey "${CERTS_DIR}/ca.key" -CAcreateserial \
  -out "${CERTS_DIR}/server.crt"

# Create Kubernetes secret
kubectl --namespace "${WEBHOOK_NAMESPACE}" create secret tls "${WEBHOOK_NAME}-tls" \
  --cert="${CERTS_DIR}/server.crt" \
  --key="${CERTS_DIR}/server.key" \
  --dry-run=client -o yaml \
  > "${SCRIPT_DIR}/manifests/webhook/webhook.tls.secret.yaml"

# Convert CA cert to base64 and indent
CA_B64=$(base64 -w0 < "${CERTS_DIR}/ca.crt")
CA_INDENTED=$(echo "$CA_B64" | fold -w 64 | sed 's/^/        /')  # 8-space indent for block under 6-space caBundle:

# Replace <CA> in the template
sed "s|<CA>|$CA_INDENTED|" "$TEMPLATE" > "$OUTPUT"

echo
echo "âœ… MutatingWebhookConfiguration written to $OUTPUT"

rm -rf "${CERTS_DIR}"
